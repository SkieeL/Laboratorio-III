// Objeto AJAX
let xhttp : XMLHttpRequest = new XMLHttpRequest();

// Abierto y enviado por método POST
xhttp.open("POST", "validar.php", true);

// Setea el encabezado de la petición (requerido para peticiones POST)
xhttp.setRequestHeader("content-type","application/x-www-form-urlencoded");

// Envía por POST con los parámetros correspondientes
xhttp.send("user=SkieL&pass=1235");

// Cuando cambia readyState se llama esta función
xhttp.onreadystatechange = () => {
    // Si el estado es 200 (ÉXITO) y la petición esta lista (4)
    if (xhttp.status == 200 && xhttp.readyState == 4) {
        // Toma la respuesta desde el servidor
        let respuesta : string = xhttp.responseText;

        // La muestra en un DIV
        (<HTMLDivElement>document.getElementById("divResultado")).innerHTML = respuesta;
    }
}
